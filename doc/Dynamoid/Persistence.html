def dump
  Hash.new.tap do |hash|
    self.class.attributes.each do |attribute, options|
      hash[attribute] = dump_field(self.read_attribute(attribute), options[:type])
    end
  end
end

def save(options = {})
  if self.new_record?
    run_callbacks(:create) do
      run_callbacks(:save) do
        persist
      end
    end
  else
    run_callbacks(:save) do
      persist
    end
  end
  self
end
